```typescript TypeScript
const res = await fetch(
  'https://api.blockworks.com/v1/metrics/treasury-crypto-asset?project=MSTR',

)
const data = await res.json()
```

```python Python
import requests

r = requests.get(
  'https://api.blockworks.com/v1/metrics/treasury-crypto-asset',

  params={'project': 'MSTR'}
)
data = r.json()
```
