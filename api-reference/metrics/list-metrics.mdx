## GET /v3/metrics

Retrieve a paginated list of available metrics that can be queried for crypto assets and protocols.

### Request

```bash
GET https://rest.staging.blockworksresearch.com/v3/metrics
```

### Authentication

<CodeGroup>

```bash cURL
curl -H "x-api-key: YOUR_API_KEY" \
  "https://rest.staging.blockworksresearch.com/v3/metrics"
```

```typescript TypeScript
const response = await fetch('https://rest.staging.blockworksresearch.com/v3/metrics', {
  headers: {
    'x-api-key': 'YOUR_API_KEY',
    'Content-Type': 'application/json'
  }
});
```

```python Python
import requests

headers = {
    'x-api-key': 'YOUR_API_KEY',
    'Content-Type': 'application/json'
}

response = requests.get(
    'https://rest.staging.blockworksresearch.com/v3/metrics',
    headers=headers
)
```

</CodeGroup>

### Query Parameters

| Parameter | Type | Description | Default |
|-----------|------|-------------|---------|
| `page` | integer | Page number (1-indexed) | 1 |
| `limit` | integer | Items per page (max 100) | 10 |
| `project` | string | Filter by project identifier | - |
| `category` | string | Filter by metric category | - |
| `search` | string | Search metrics by name or description | - |

### Response Fields

| Field | Type | Description |
|-------|------|-------------|
| `name` | string | Human-readable metric name |
| `description` | string | Detailed metric description |
| `identifier` | string | Unique metric identifier |
| `project` | string | Project this metric belongs to |
| `source` | string | Data source (e.g., "Blockworks") |
| `data_type` | string | Data type (e.g., "timeseries_int", "timeseries_usd") |
| `parameters` | object | Available query parameters |
| `updated_at` | number | Unix timestamp of last update |
| `total` | number | Total number of metrics available |
| `page` | number | Current page number |

### Metric Categories

The API organizes metrics into several categories:

| Category | Description | Examples |
|----------|-------------|----------|
| `defi` | DeFi protocol metrics | TVL, fees, liquidity |
| `trading` | Trading-related metrics | Volume, spreads, trades |
| `network` | Network activity metrics | Users, transactions, gas |
| `token` | Token-specific metrics | Supply, distribution, holders |
| `governance` | Governance metrics | Proposals, voting, participation |

### Common Use Cases

#### Building a Metric Dashboard

Get metrics for a comprehensive dashboard:

```typescript
async function buildMetricDashboard() {
  const categories = ['defi', 'trading', 'network'];
  const dashboardMetrics = {};
  
  for (const category of categories) {
    const response = await fetch(`https://rest.staging.blockworksresearch.com/v3/metrics?category=${category}&limit=20`, {
      headers: { 'x-api-key': 'YOUR_API_KEY' }
    });
    
    const data = await response.json();
    dashboardMetrics[category] = data.data;
  }
  
  return dashboardMetrics;
}

// Usage
const dashboard = await buildMetricDashboard();
console.log('Dashboard metrics:', dashboard);
```

#### Finding Project-Specific Metrics

Get all available metrics for a specific project:

```python
def get_project_metrics(project_id: str):
    """Get all metrics available for a specific project"""
    headers = {'x-api-key': 'YOUR_API_KEY'}
    params = {'project': project_id, 'limit': 100}
    
    response = requests.get(
        'https://rest.staging.blockworksresearch.com/v3/metrics',
        headers=headers,
        params=params
    )
    
    if response.status_code == 200:
        data = response.json()
        metrics = data['data']
        
        print(f"Available metrics for {project_id}:")
        for metric in metrics:
            print(f"  {metric['identifier']}: {metric['name']}")
            print(f"    Unit: {metric['unit']}")
            print(f"    Frequency: {metric['frequency']}")
            print(f"    Available from: {metric['data_availability']['start_date']}")
            print()
        
        return metrics
    
    return None

# Usage
uniswap_metrics = get_project_metrics('uniswap')
ethereum_metrics = get_project_metrics('ethereum')
```

#### Metric Discovery

Discover relevant metrics based on keywords:

```typescript
async function discoverMetrics(keywords: string[]) {
  const discoveredMetrics = [];
  
  for (const keyword of keywords) {
    const response = await fetch(`https://rest.staging.blockworksresearch.com/v3/metrics?search=${keyword}`, {
      headers: { 'x-api-key': 'YOUR_API_KEY' }
    });
    
    const data = await response.json();
    discoveredMetrics.push({
      keyword,
      metrics: data.data
    });
  }
  
  return discoveredMetrics;
}

// Usage
const keywords = ['volume', 'fees', 'users', 'liquidity'];
const discovered = await discoverMetrics(keywords);

discovered.forEach(({ keyword, metrics }) => {
  console.log(`\n${keyword.toUpperCase()} METRICS:`);
  metrics.forEach(metric => {
    console.log(`  ${metric.identifier}: ${metric.name}`);
  });
});
```

#### Metric Availability Analysis

Analyze data availability across metrics:

```python
def analyze_metric_availability():
    """Analyze data availability across all metrics"""
    headers = {'x-api-key': 'YOUR_API_KEY'}
    params = {'limit': 100}
    
    response = requests.get(
        'https://rest.staging.blockworksresearch.com/v3/metrics',
        headers=headers,
        params=params
    )
    
    if response.status_code == 200:
        data = response.json()
        metrics = data['data']
        
        # Group by category
        categories = {}
        for metric in metrics:
            category = metric['category']
            if category not in categories:
                categories[category] = []
            categories[category].append(metric)
        
        # Analyze each category
        for category, cat_metrics in categories.items():
            print(f"\n{category.upper()} METRICS:")
            print(f"  Count: {len(cat_metrics)}")
            
            # Find date ranges
            start_dates = [m['data_availability']['start_date'] for m in cat_metrics]
            end_dates = [m['data_availability']['end_date'] for m in cat_metrics]
            
            print(f"  Earliest data: {min(start_dates)}")
            print(f"  Latest data: {max(end_dates)}")
            
            # Count projects
            all_projects = set()
            for metric in cat_metrics:
                all_projects.update(metric['projects'])
            print(f"  Supported projects: {len(all_projects)}")
    
    return None

# Usage
analyze_metric_availability()
```

### How to Query Metrics

```typescript

});
```
