## GET /v3/metrics/{identifier}

Retrieve time-series data for a specific metric, with optional filtering by project and time range.

### Request

```bash
GET https://rest.staging.blockworksresearch.com/v3/metrics/{identifier}
```

### Path Parameters

| Parameter | Type | Description | Required |
|-----------|------|-------------|----------|
| `identifier` | string | Unique metric identifier | Yes |

### Authentication

<CodeGroup>

```bash cURL
curl -H "x-api-key: YOUR_API_KEY" \
  "https://rest.staging.blockworksresearch.com/v3/metrics/tvl"
```

```typescript TypeScript
const response = await fetch('https://rest.staging.blockworksresearch.com/v3/metrics/tvl', {
  headers: {
    'x-api-key': 'YOUR_API_KEY',
    'Content-Type': 'application/json'
  }
});
```

```python Python
import requests

headers = {
    'x-api-key': 'YOUR_API_KEY',
    'Content-Type': 'application/json'
}

response = requests.get(
    'https://rest.staging.blockworksresearch.com/v3/metrics/tvl',
    headers=headers
)
```

</CodeGroup>

### Query Parameters

| Parameter | Type | Description | Default |
|-----------|------|-------------|---------|
| `project` | string | Filter by project identifier | - |
| `start_date` | string | Start date (YYYY-MM-DD) | - |
| `end_date` | string | End date (YYYY-MM-DD) | - |

<Warning>
**Note**: The `limit`, `page`, and `sort` parameters are NOT supported by the metrics query endpoint.
</Warning>

curl -H "x-api-key: YOUR_API_KEY" \

```typescript TypeScript
async function compareProjects(metric: string, projects: string[]) {
  const results = {};
  
  for (const project of projects) {
    const response = await fetch(`https://rest.staging.blockworksresearch.com/v3/metrics/${metric}?project=${project}&limit=30`, {
      headers: { 'x-api-key': 'YOUR_API_KEY' }
    });
    
    const data = await response.json();
    results[project] = data.data;
  }
  
  return results;
}

// Usage
const projects = ['uniswap', 'sushiswap', 'curve'];
const volumeComparison = await compareProjects('volume_24h', projects);

// Calculate average volume for each project
Object.entries(volumeComparison).forEach(([project, data]) => {
  const avgVolume = data.reduce((sum, point) => sum + point.value, 0) / data.length;
  console.log(`${project}: Average 24h volume = $${avgVolume.toLocaleString()}`);
});
```

```python Python
import requests

def compare_projects(metric: str, projects: list):
    """Compare metric data across multiple projects"""
    results = {}
    
    for project in projects:
        headers = {'x-api-key': 'YOUR_API_KEY'}
        params = {'project': project, 'limit': 30}
        
        response = requests.get(
            f'https://rest.staging.blockworksresearch.com/v3/metrics/{metric}',
            headers=headers,
            params=params
        )
        
        if response.status_code == 200:
            data = response.json()
            results[project] = data['data']
    
    return results

# Usage
projects = ['uniswap', 'sushiswap', 'curve']
volume_comparison = compare_projects('volume_24h', projects)

# Calculate average volume for each project
for project, data in volume_comparison.items():
    avg_volume = sum(point['value'] for point in data) / len(data)
    print(f"{project}: Average 24h volume = ${avg_volume:,.2f}")
```

### Response Fields

The response is structured as a JSON object where the key is the project name and the value is an array of data points:

| Field | Type | Description |
|-------|------|-------------|
| `{ProjectName}` | array | Array of metric data points for the project |
| `date` | string | Date in YYYY-MM-DD format |
| `value` | number | Metric value for that date |

<Note>
The response does not include pagination information. All available data points are returned.
</Note>

### Common Use Cases

#### Time Series Analysis

Analyze metric trends over time:

```typescript
async function analyzeMetricTrend(metric: string, project: string, days: number = 30) {
  const endDate = new Date();
  const startDate = new Date(endDate);
  startDate.setDate(startDate.getDate() - days);
  
  const response = await fetch(`https://rest.staging.blockworksresearch.com/v3/metrics/${metric}?project=${project}&start_date=${startDate.toISOString().split('T')[0]}&end_date=${endDate.toISOString().split('T')[0]}&sort=asc`, {
    headers: { 'x-api-key': 'YOUR_API_KEY' }
  });
  
  const data = await response.json();
  const points = data.data;
  
  if (points.length < 2) return null;
  
  // Calculate trend
  const firstValue = points[0].value;
  const lastValue = points[points.length - 1].value;
  const change = ((lastValue - firstValue) / firstValue) * 100;
  
  // Find min/max
  const minPoint = points.reduce((min, point) => point.value < min.value ? point : min);
  const maxPoint = points.reduce((max, point) => point.value > max.value ? point : max);
  
  return {
    project,
    metric,
    period: `${days} days`,
    change: `${change.toFixed(2)}%`,
    current: lastValue,
    min: { value: minPoint.value, date: minPoint.timestamp },
    max: { value: maxPoint.value, date: maxPoint.timestamp }
  };
}

// Usage
const trend = await analyzeMetricTrend('tvl', 'uniswap', 30);
console.log('Trend analysis:', trend);
```

#### Portfolio Metrics Dashboard

Build a comprehensive metrics dashboard:

```python
import requests
import pandas as pd
from datetime import datetime, timedelta

def build_portfolio_dashboard(projects: list, metrics: list):
    """Build a comprehensive metrics dashboard"""
    dashboard_data = {}
    
    for project in projects:
        dashboard_data[project] = {}
        
        for metric in metrics:
            try:
                headers = {'x-api-key': 'YOUR_API_KEY'}
                params = {'project': project, 'limit': 7}  # Last 7 days
                
                response = requests.get(
                    f'https://rest.staging.blockworksresearch.com/v3/metrics/{metric}',
                    headers=headers,
                    params=params
                )
                
                if response.status_code == 200:
                    data = response.json()
                    points = data['data']
                    
                    if points:
                        # Calculate summary statistics
                        values = [point['value'] for point in points]
                        dashboard_data[project][metric] = {
                            'current': values[0],  # Most recent
                            'average': sum(values) / len(values),
                            'min': min(values),
                            'max': max(values),
                            'trend': 'up' if values[0] > values[-1] else 'down',
                            'data_points': len(values)
                        }
                
            except Exception as e:
                print(f"Error fetching {metric} for {project}: {e}")
    
    return dashboard_data

# Usage
projects = ['uniswap', 'aave', 'compound']
metrics = ['tvl', 'volume_24h', 'daily_active_users']
dashboard = build_portfolio_dashboard(projects, metrics)

# Display dashboard
for project, project_metrics in dashboard.items():
    print(f"\n{project.upper()} METRICS:")
    for metric, stats in project_metrics.items():
        print(f"  {metric}:")
        print(f"    Current: {stats['current']:,.2f}")
        print(f"    Average: {stats['average']:,.2f}")
        print(f"    Trend: {stats['trend']}")
```

#### Historical Data Analysis

Analyze long-term historical trends:

```typescript
async function analyzeHistoricalData(metric: string, project: string, months: number = 12) {
  const endDate = new Date();
  const startDate = new Date(endDate);
  startDate.setMonth(startDate.getMonth() - months);
  
  const response = await fetch(`https://rest.staging.blockworksresearch.com/v3/metrics/${metric}?project=${project}&start_date=${startDate.toISOString().split('T')[0]}&end_date=${endDate.toISOString().split('T')[0]}&limit=1000&sort=asc`, {
    headers: { 'x-api-key': 'YOUR_API_KEY' }
  });
  
  const data = await response.json();
  const points = data.data;
  
  // Group by month
  const monthlyData = {};
  points.forEach(point => {
    const month = point.timestamp.substring(0, 7); // YYYY-MM
    if (!monthlyData[month]) {
      monthlyData[month] = [];
    }
    monthlyData[month].push(point.value);
  });
  
  // Calculate monthly averages
  const monthlyAverages = Object.entries(monthlyData).map(([month, values]) => ({
    month,
    average: values.reduce((sum, val) => sum + val, 0) / values.length,
    min: Math.min(...values),
    max: Math.max(...values),
    dataPoints: values.length
  }));
  
  return {
    project,
    metric,
    period: `${months} months`,
    monthlyAverages,
    totalDataPoints: points.length
  };
}

// Usage
const analysis = await analyzeHistoricalData('tvl', 'uniswap', 12);
console.log('Historical analysis:', analysis);
```

#### Real-time Monitoring

Set up real-time monitoring for key metrics:

```python
import requests
import time
from datetime import datetime

def monitor_metrics(metrics_config: dict, interval: int = 300):
    """Monitor metrics in real-time"""
    while True:
        print(f"\n=== METRICS MONITOR - {datetime.now().strftime('%Y-%m-%d %H:%M:%S')} ===")
        
        for metric, projects in metrics_config.items():
            print(f"\n{metric.upper()}:")
            
            for project in projects:
                try:
                    headers = {'x-api-key': 'YOUR_API_KEY'}
                    params = {'project': project, 'limit': 1}
                    
                    response = requests.get(
                        f'https://rest.staging.blockworksresearch.com/v3/metrics/{metric}',
                        headers=headers,
                        params=params
                    )
                    
                    if response.status_code == 200:
                        data = response.json()
                        if data['data']:
                            point = data['data'][0]
                            print(f"  {project}: {point['value']:,.2f} ({point['timestamp']})")
                
                except Exception as e:
                    print(f"  {project}: Error - {e}")
        
        time.sleep(interval)

# Usage
metrics_config = {
    'tvl': ['uniswap', 'aave', 'compound'],
    'volume_24h': ['uniswap', 'sushiswap'],
    'daily_active_users': ['ethereum', 'polygon']
}

# Monitor every 5 minutes
monitor_metrics(metrics_config, interval=300)
```

### Understanding Metric Frequencies

Different metrics have different update frequencies:

| Frequency | Description | Examples |
|-----------|-------------|----------|
| `real-time` | Updated continuously | Price feeds, trading volume |
| `hourly` | Updated every hour | Gas prices, transaction counts |
| `daily` | Updated once per day | TVL, daily active users |
| `weekly` | Updated weekly | Governance metrics |

### Rate Limiting
